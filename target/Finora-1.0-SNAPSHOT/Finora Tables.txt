CREATE TABLE USER_DETAILS (
    NAME VARCHAR(100),
    EMAIL VARCHAR(100),
    PHONE VARCHAR(20) PRIMARY KEY,
    PAN VARCHAR(20),
    PASSWORD VARCHAR(50),
    ACCOUNT_NO BIGINT UNIQUE,
    SORT_CODE VARCHAR(10) DEFAULT '40-02-61',
    BALANCE DOUBLE DEFAULT 0.0,
    NOMINEE_NAME VARCHAR(100) DEFAULT 'Not Updated Yet',
    NOMINEE_RELATION VARCHAR(50) DEFAULT 'Not Updated Yet',
    REWARDS INT DEFAULT 0,
    ACTIVE_LOAN DOUBLE DEFAULT 0.0,
    USED_LOAN_LIMIT DOUBLE DEFAULT 0.0,
    STATUS VARCHAR(20) DEFAULT 'Pending'
);

CREATE TABLE TRANSACTIONS (
    TID INT GENERATED ALWAYS AS IDENTITY,
    SENDER_ACC BIGINT,
    RECEIVER_ACC BIGINT,
    AMOUNT DOUBLE,
    TDATE TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    REMARKS VARCHAR(100),
    ORIGINAL_AMOUNT DOUBLE DEFAULT 0.0
);

CREATE TABLE BENEFICIARIES (
    BID INT NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
    OWNER_PHONE VARCHAR(20),
    BENEFICIARY_ACC BIGINT,
    BENEFICIARY_NAME VARCHAR(100),
    PRIMARY KEY (BID)
);

CREATE TABLE SUPPORT_TICKETS (
    TICKET_ID INT NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 101, INCREMENT BY 1),
    USER_PHONE VARCHAR(20),
    SUBJECT VARCHAR(100),
    ISSUE_TYPE VARCHAR(50),
    DESCRIPTION VARCHAR(500),
    STATUS VARCHAR(20) DEFAULT 'OPEN',
    CREATED_AT TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (TICKET_ID),
    ADMIN_REPLY VARCHAR(500) DEFAULT 'No reply yet'
);
